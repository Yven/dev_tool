### 3.3 commands/timestamp.rs — 时间戳转换

```rust
use chrono::{DateTime, Local, TimeZone, Utc};
use serde::Serialize;

#[derive(Serialize)]
pub struct TimestampResult {
    seconds: i64,
    millis: i64,
    micros: i64,
    rfc3339: String,
    iso8601: String,
    excel: f64,
    local: String,
}

#[tauri::command]
pub fn cmd_timestamp_convert(input: &str, unit: &str, timezone: &str) -> Result<TimestampResult, String> {
    let value: i64 = input.parse().map_err(|e| format!("无法解析数字: {}", e))?;

    // 归一化到微秒
    let micros = match unit {
        "s" => value * 1_000_000,
        "ms" => value * 1_000,
        "us" => value,
        _ => return Err(format!("未知单位: {}", unit)),
    };

    let seconds = micros / 1_000_000;
    let millis = micros / 1_000;

    let dt: DateTime<Utc> = Utc.timestamp_opt(seconds, 0).single()
        .ok_or_else(|| "时间戳超出范围".to_string())?;

    let local_dt = if timezone == "UTC" { dt.with_timezone(&Utc) } else { dt.with_timezone(&Local) };

    // Excel 日期序列号 (1900-01-01 = 1)
    let excel = (seconds as f64 - -2209161600.0) / 86400.0 + 2.0;

    Ok(TimestampResult {
        seconds,
        millis,
        micros,
        rfc3339: dt.to_rfc3339(),
        iso8601: dt.format("%Y-%m-%dT%H:%M:%S").to_string(),
        excel,
        local: local_dt.format("%Y-%m-%d %H:%M:%S %:z").to_string(),
    })
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_seconds_to_millis() {
        let r = cmd_timestamp_convert("1609459200", "s", "UTC").unwrap();
        assert_eq!(r.millis, 1609459200000);
        assert_eq!(r.micros, 1609459200000000);
    }

    #[test]
    fn test_invalid_input() {
        assert!(cmd_timestamp_convert("abc", "s", "UTC").is_err());
    }
}
```

### 3.4 commands/encoding.rs — UTF-8 中文转换

```rust
use serde::Serialize;

#[derive(Serialize)]
pub struct EncodingResult {
    output: String,
    detected_type: String,
}

/// 自动检测并转换
#[tauri::command]
pub fn cmd_encoding_convert(input: &str, direction: &str) -> Result<EncodingResult, String> {
    let detected = detect_encoding_type(input);

    let output = match direction {
        "decode" => decode_input(input, &detected)?,
        "encode" => encode_input(input)?,
        _ => return Err("direction 必须为 decode 或 encode".into()),
    };

    Ok(EncodingResult { output, detected_type: detected })
}

fn detect_encoding_type(s: &str) -> String {
    if s.contains("\\u") { "unicode_escape".into() }
    else if s.contains("&#x") || s.contains("&#X") { "html_entity".into() }
    else if s.contains('%') && s.chars().filter(|c| *c == '%').count() >= 2 { "percent".into() }
    else { "plain".into() }
}

fn decode_input(input: &str, enc_type: &str) -> Result<String, String> {
    match enc_type {
        "unicode_escape" => decode_unicode_escape(input),
        "html_entity" => decode_html_entity(input),
        "percent" => decode_percent(input),
        _ => Ok(input.into()),
    }
}

fn decode_unicode_escape(s: &str) -> Result<String, String> {
    let re = regex::Regex::new(r"\\u([0-9a-fA-F]{4})").map_err(|e| e.to_string())?;
    let mut result = s.to_string();
    for cap in re.captures_iter(s) {
        let code: u32 = u32::from_str_radix(&cap[1], 16).map_err(|e| e.to_string())?;
        if let Some(ch) = char::from_u32(code) {
            result = result.replace(&cap[0], &ch.to_string());
        }
    }
    Ok(result)
}
```

### 3.5 commands/url.rs — URL 编解码

```rust
use percent_encoding::{utf8_percent_encode, NON_ALPHANUMERIC, AsciiSet};
use serde::Serialize;

const RFC3986: &AsciiSet = &NON_ALPHANUMERIC
    .remove(b'-').remove(b'_').remove(b'.').remove(b'~');

#[derive(Serialize)]
pub struct UrlResult {
    output: String,
    line_count: usize,
}

#[tauri::command]
pub fn cmd_url_encode(input: &str, safe_chars: Option<String>) -> Result<UrlResult, String> {
    let lines: Vec<&str> = input.lines().collect();
    let encoded: Vec<String> = lines.iter().map(|line| {
        utf8_percent_encode(line, RFC3986).to_string()
    }).collect();
    Ok(UrlResult { output: encoded.join("\n"), line_count: lines.len() })
}

#[tauri::command]
pub fn cmd_url_decode(input: &str) -> Result<UrlResult, String> {
    let lines: Vec<&str> = input.lines().collect();
    let decoded: Vec<String> = lines.iter().map(|line| {
        percent_encoding::percent_decode_str(line).decode_utf8_lossy().to_string()
    }).collect();
    Ok(UrlResult { output: decoded.join("\n"), line_count: lines.len() })
}
```

### 3.6 commands/json_fmt.rs — JSON 格式化

```rust
use serde::Serialize;
use serde_json::Value;

#[derive(Serialize)]
pub struct JsonResult {
    output: String,
    error_line: Option<usize>,
    error_msg: Option<String>,
}

#[tauri::command]
pub fn cmd_json_format(
    input: &str,
    indent: u32,
    sort_keys: bool,
    escape_unicode: bool,
    compress: bool,
) -> Result<JsonResult, String> {
    // 先尝试解析
    let mut value: Value = serde_json::from_str(input).map_err(|e| {
        // 计算错误行号
        let line = e.line();
        JsonResult {
            output: String::new(),
            error_line: Some(line),
            error_msg: Some(e.to_string()),
        }.to_string() // 通过 serialize 传递
    }).map_err(|e| format!("JSON解析失败: {}", e))?;

    // 键名排序
    if sort_keys {
        sort_json_object(&mut value);
    }

    let output = if compress {
        serde_json::to_string(&value).map_err(|e| e.to_string())?
    } else {
        let indent_str = " ".repeat(indent as usize);
        if escape_unicode {
            // Unicode 转义模式
            let s = serde_json::to_string_pretty(&value).map_err(|e| e.to_string())?;
            escape_unicode_chars(&s)
        } else {
            serde_json::to_string_pretty(&value).map_err(|e| e.to_string())?
        }
    };

    Ok(JsonResult { output, error_line: None, error_msg: None })
}

fn sort_json_object(v: &mut Value) {
    if let Value::Object(map) = v {
        let mut sorted = serde_json::Map::new();
        let mut keys: Vec<_> = map.keys().cloned().collect();
        keys.sort();
        for k in keys {
            let mut val = map.remove(&k).unwrap();
            sort_json_object(&mut val);
            sorted.insert(k, val);
        }
        *map = sorted;
    } else if let Value::Array(arr) = v {
        for item in arr.iter_mut() { sort_json_object(item); }
    }
}

fn escape_unicode_chars(s: &str) -> String {
    s.chars().map(|c| {
        if c as u32 > 127 {
            format!("\\u{:04x}", c as u32)
        } else {
            c.to_string()
        }
    }).collect()
}
```

### 3.8 window/edge_hide.rs — 贴边隐藏

```rust
use tauri::{AppHandle, Manager, WebviewWindow};
use std::sync::Mutex;

pub enum EdgeSide { Top, Left, Right, None }

pub struct EdgeHideState {
    pub current_edge: Mutex<EdgeSide>,
    pub is_collapsed: Mutex<bool>,
    pub full_size: Mutex<(u32, u32)>,  // 记忆完整窗口尺寸
}

const TRIGGER_WIDTH: u32 = 3;

pub fn detect_edge(window: &WebviewWindow) -> EdgeSide {
    if let (Ok(pos), Ok(size), Ok(monitor)) = (
        window.outer_position(),
        window.outer_size(),
        window.current_monitor(),
    ) {
        let mon = monitor.unwrap();
        let mon_pos = mon.position();
        let mon_size = mon.size();

        let x = pos.x;
        let y = pos.y;

        // 上边吸附
        if y <= mon_pos.y + 5 { return EdgeSide::Top; }
        // 左边吸附
        if x <= mon_pos.x + 5 { return EdgeSide::Left; }
        // 右边吸附
        if x + size.width as i32 >= mon_pos.x + mon_size.width as i32 - 5 { return EdgeSide::Right; }
    }
    EdgeSide::None
}

pub fn collapse_to_trigger(window: &WebviewWindow, edge: &EdgeSide, full_size: (u32, u32)) {
    match edge {
        EdgeSide::Top => {
            let _ = window.set_size(tauri::PhysicalSize::new(full_size.0, TRIGGER_WIDTH));
        }
        EdgeSide::Left => {
            let _ = window.set_size(tauri::PhysicalSize::new(TRIGGER_WIDTH, full_size.1));
        }
        EdgeSide::Right => {
            let _ = window.set_size(tauri::PhysicalSize::new(TRIGGER_WIDTH, full_size.1));
        }
        EdgeSide::None => {}
    }
}

pub fn expand_full(window: &WebviewWindow, full_size: (u32, u32)) {
    let _ = window.set_size(tauri::PhysicalSize::new(full_size.0, full_size.1));
}
```

### 3.9 plugin_loader/ — 插件扫描器

```rust
// scanner.rs
use std::fs;
use std::path::PathBuf;
use tauri::{App, Manager};

const MAX_PLUGIN_SIZE: u64 = 200 * 1024; // 200KB

pub struct PluginMeta {
    pub id: String,
    pub name: String,
    pub version: String,
    pub file_path: PathBuf,
}

pub fn scan_plugins_dir(app: &App) -> Result<Vec<PluginMeta>, String> {
    let plugins_dir = app.path().app_data_dir().unwrap().join("plugins");
    if !plugins_dir.exists() { return Ok(vec![]); }

    let mut plugins = Vec::new();
    for entry in fs::read_dir(&plugins_dir).map_err(|e| e.to_string())? {
        let entry = entry.map_err(|e| e.to_string())?;
        let path = entry.path();

        if !path.extension().map(|e| e == "js").unwrap_or(false) { continue; }

        // 文件大小检查
        let metadata = fs::metadata(&path).map_err(|e| e.to_string())?;
        if metadata.len() > MAX_PLUGIN_SIZE { continue; }

        // ESModule 检查: 文件头包含 export
        let content = fs::read_to_string(&path).map_err(|e| e.to_string())?;
        if !content.contains("export default") { continue; }

        plugins.push(PluginMeta {
            id: path.file_stem().unwrap().to_string_lossy().to_string(),
            name: path.file_stem().unwrap().to_string_lossy().to_string(),
            version: "0.0.0".into(),
            file_path: path,
        });
    }
    Ok(plugins)
}

pub fn start_watch(app: &App) -> Result<(), Box<dyn std::error::Error>> {
    // 使用 notify crate 监听 plugins/ 目录变化
    // 文件新增 → 调用 scan + emit "plugin-loaded" 事件
    // 文件删除 → emit "plugin-unloaded" 事件
    // v0.1 先用定时轮询，后续优化为 fs watcher
    Ok(())
}
```

### 4.6 modules/json/JsonPanel.tsx — JSON 格式化面板

```tsx
import { createSignal } from 'solid-js';
import { invoke } from '@tauri-apps/api/core';
import InputArea from '../../components/InputArea';
import OutputArea from '../../components/OutputArea';

interface JsonResult {
  output: string;
  error_line: number | null;
  error_msg: string | null;
}

export default function JsonPanel() {
  const [input, setInput] = createSignal('');
  const [indent, setIndent] = createSignal(2);
  const [sortKeys, setSortKeys] = createSignal(false);
  const [escapeUnicode, setEscapeUnicode] = createSignal(false);
  const [compress, setCompress] = createSignal(false);

  const format = async () => {
    if (!input().trim()) return;
    try {
      const r = await invoke<JsonResult>('cmd_json_format', {
        input: input(), indent: indent(), sortKeys: sortKeys(),
        escapeUnicode: escapeUnicode(), compress: compress(),
      });
      setResult(r);
    } catch (e) { setResult({ output: '', error_line: null, error_msg: String(e) }); }
  };

  return (
    <div class="flex flex-col gap-3">
      <InputArea value={input} onInput={setInput} placeholder='输入 JSON...' />
      <div class="flex gap-2 flex-wrap">
        <button class="px-2 py-1 text-xs rounded border border-border hover:border-primary"
          classList={{ 'bg-primary text-white': indent() === 2 }} onclick={() => setIndent(2)}>2空格</button>
```

### 4.7 plugin/sdk.ts — 插件 SDK

```typescript
import { readText, writeText } from '@tauri-apps/plugin-clipboard-manager';
import { invoke } from '@tauri-apps/api/core';

export interface PluginSDK {
  clipboard: {
    readText(): Promise<string>;
    writeText(text: string): Promise<void>;
  };
  storage: {
    get(key: string): Promise<string | null>;
    set(key: string, value: string): Promise<void>;
  };
  invoke(cmd: string, args?: Record<string, unknown>): Promise<unknown>;
}

export function createPluginSDK(pluginId: string): PluginSDK {
  return {
    clipboard: { readText, writeText },
    storage: {
      async get(key: string) {
        try {
          return await invoke<string>('plugin_storage_get', { pluginId, key });
        } catch { return null; }
      },
      async set(key: string, value: string) {
        await invoke('plugin_storage_set', { pluginId, key, value });
      },
    },
    invoke: (cmd, args) => invoke(cmd, args),
  };
}
```

### 4.8 plugin/PluginSlot.tsx — 插件 UI 插槽

```tsx
import { onMount, onCleanup } from 'solid-js';
import { createPluginSDK } from './sdk';

interface Props { id: string; }

export default function PluginSlot(props: Props) {
  let containerRef: HTMLDivElement | undefined;
  let pluginDestroy: (() => void) | undefined;

  onMount(async () => {
    if (!containerRef) return;
    const sdk = createPluginSDK(props.id);
    // 动态加载插件 JS
    const module = await import(`/plugins/${props.id.replace('plugin:', '')}.js`);
    const plugin = module.default;
    plugin.render(containerRef, sdk);
    pluginDestroy = plugin.destroy;
  });

  onCleanup(() => { pluginDestroy?.(); });

  return <div ref={containerRef} class="w-full h-full" />;
}
```
